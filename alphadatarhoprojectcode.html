            class BloodPressureMonitor():
            
              ####defines inputs. When inputting objects, need systolic value, then diastolic value, then BPM device ID, then first name, then lastname
                def __init__(self, systolic, diastolic, BPMdeviceID, firstname, lastname):
            
                    self.systolic = int(systolic)
                    self.diastolic = int(diastolic)
                    self.BPMdeviceID = int(BPMdeviceID)
                    self.firstname = firstname
                    self.lastname = lastname
            
                ####function that measures blood pressure
                def measureBP(self):
                    if self.systolic < 120 and self.diastolic < 80:
                        print(self.systolic)
                        print(self.diastolic)
                        print ("Normal BP")
                    elif self.systolic in range(120,129) and self.diastolic < 80:
                        print(self.systolic)
                        print(self.diastolic)
                        print("Elevated BP")
                    elif self.systolic in range(130,139) or self.diastolic in range(80,89):
                        print(self.systolic)
                        print(self.diastolic)
                        print("High BP")
                    elif self.systolic >= 140 or self.diastolic >= 90:
                        print(self.systolic)
                        print(self.diastolic)
                        print("Very High BP")
            
                ####function that displays blood pressure
                def displayBP(self):
                    self.measureBP()
            
                ##### function that displays device information
                def displaydeviceinfo(self):
                    print(self.BPMdeviceID)
                    print("assigned to:", self.firstname, self.lastname)
